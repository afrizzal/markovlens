----- BEGIN ci.yml CONTENT -----
name: CI

on:
  push:
    branches: [master]
  pull_request:
    branches: [master]

permissions:
  contents: read

jobs:
  test:
    name: Lint, type-check, and test (Python 3.12)
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Install uv and set up Python 3.12
        uses: astral-sh/setup-uv@v8
        with:
          python-version: "3.12"
          enable-cache: true

      - name: Install dependencies (clean-room uv.lock rehearsal)
        run: uv sync

      - name: Ruff lint
        run: uv run ruff check .

      - name: Ruff format check
        run: uv run ruff format --check .

      - name: Mypy type check
        run: uv run mypy core/ domains/

      - name: Pytest (excluding slow)
        run: uv run pytest -m "not slow"
----- END ci.yml CONTENT -----

Write the file with exactly the lines between the BEGIN/END markers (do not include the marker lines themselves).
