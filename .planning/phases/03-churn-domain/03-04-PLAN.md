```python
baseline_P = result.transition_matrix
overrides: dict[tuple[int, int], float] = {}
any_changed = False
with left:
    st.markdown('<div class="t-h3">Adjust transition probabilities</div>'
                '<div class="t-xs t-ter">Each row auto-renormalizes to 100%. Ghost value = baseline.</div>',
                unsafe_allow_html=True)
    for i, from_label in enumerate(result.state_labels):
        with st.expander(f"From {from_label}", expanded=(i == 0)):
            for j, to_label in enumerate(result.state_labels):
                baseline_val = float(baseline_P[i, j])
                key = f"{PAGE_NS}.what_if.{i}_{j}"   # namespaced (Pitfall 3)
                if key not in st.session_state:
                    st.session_state[key] = baseline_val
                current = st.slider(f"→ {to_label}  (baseline {baseline_val:.2f})",
                                    0.0, 1.0, step=0.01, format="%.2f", key=key)
                if abs(current - baseline_val) > 1e-6:
                    overrides[(i, j)] = current
                    any_changed = True
    if any_changed:
        if st.button("Reset all", key=f"{PAGE_NS}.reset"):
            for i in range(len(result.state_labels)):
                for j in range(len(result.state_labels)):
                    st.session_state.pop(f"{PAGE_NS}.what_if.{i}_{j}", None)
            st.rerun()
```
Note: do NOT pass `value=` to st.slider when `key=` is set and already initialized in session_state (Streamlit warns on value+key collision); initialization via `st.session_state[key] = baseline_val` is the single source of truth.
